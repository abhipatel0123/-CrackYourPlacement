 long long findMinDiff(vector<long long> a, long long n, long long m){
    //code
        if(a.size() < m){
            return 0;
        }
           sort(a.begin(),a.end());
           int min=INT_MAX;
           for(int i=0;i<=a.size()-m;i++){
               int diff =a[i+m-1]-a[i];
               if(diff<min ){
                   min=diff;
               }
           }
           return min;
    
    }
